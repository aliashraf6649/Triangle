Insertion_Sort(A,n)
	//Array A of size n is passed
	for j ← 1 to n
		key ← A[j]
		i ← j - 1
		while i >= 0 && A[i] > key
			do
				A[i+1] = A[i]
				i ← i-1
		A[i+1] = key
	return A
	
isTriangular(A,n)
	//Array A of size n is passed
	Insertion_Sort(A)
	
	for i ← to n-2
		if A[i] + A[i + 1] > A[i + 2] && A[i + 1] + A[i+2] > A[i] && A[i] + A[i+2] > A [i+1]
			return 1
		else
			return 0
